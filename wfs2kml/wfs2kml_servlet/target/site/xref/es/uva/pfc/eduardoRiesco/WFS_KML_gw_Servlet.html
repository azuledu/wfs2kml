155 				kmlout.write("<name>"+typeName+"</name>\n"); 
156 				kmlout.write("<open>1</open>\n");
157 				kmlout.write("<description>test file</description>\n");

212 		private void kmlPlacemarks(FeatureCollection featureCollection, CoordinateReferenceSystem geomCRS) {
213 			Iterator iterator = featureCollection.iterator();			// Feature
214 	        try {
215 	            for( int f=0; iterator.hasNext(); f++) {
216 	                Feature feature = (Feature) iterator.next();
217 					kmlout.write("\n<Placemark>\n");  						
218 					kmlout.write("<name>"+feature.getID() +"</name>\n");
219 					kmlout.write("<styleUrl>default</styleUrl>\n");
220 					
221 					kmlGeometries( feature, geomCRS );
222 					
223 					kmlout.write("</Placemark>\n");
224 	            }
225 	        }
226 	        finally {
227 	            	featureCollection.close( iterator );
228 	        }

232 		 * @param feature	Feature to extract their geometries
233 		 */
234 		private void kmlGeometries(Feature feature, CoordinateReferenceSystem geomCRS) {
235 			
236 			boolean multiGeometry = false;
237 			double zCoord;
238 			CoordinateReferenceSystem kmlCRS;
239 			
240 			try {
241 				
242 			kmlCRS = CRS.decode("EPSG:4326");
243 			
244 			if (feature.getPrimaryGeometry().getNumGeometries()>1){	
245 				kmlout.write("<MultiGeometry>\n");
246 				multiGeometry = true;
247 			}
248 			for(int g=0; g<feature.getPrimaryGeometry().getNumGeometries(); g++){  // Geometry
249 				kmlout.write("<Polygon>\n");					// 	TODO geometrias diferentes de "poligon" 
250 				kmlout.write("<extrude>1</extrude>\n");
251 				//kmlout.write("<tessellate>1</tessellate>\n");  	TODO Activar para poligonos grandes.
252 				kmlout.write("<altitudeMode>relativeToGround</altitudeMode>\n");	// TODO Otros altitudeMode
253 				
254 				kmlout.write("<outerBoundaryIs>\n");  // TODO prever InnerBoundaryIS
255 				
256 				kmlout.write("<LinearRing>\n");
257 				kmlout.write("<coordinates>\n");
258 				// JTSUtilities
259 				
260 				// zMinMax(Coordinate[] cs)
261 		        //  Determine the min and max "z" values in an array of Coordinates.
262 				
263 				// guessCoorinateDims(Coordinate[] cs)
264 //		          Returns:
265 //		        	  2 for 2d (default)
266 //		        	  4 for 3d - one of the oordinates has a non-NaN z value
267 //		        	  (3 is for x,y,m but thats not supported yet) 
268 				
269 //				findBestGeometryType(Geometry geom)
270 //		          Determine the best ShapeType for a given Geometry.
271 				
272 				Geometry geomGeometry = feature.getPrimaryGeometry().getGeometryN(g);
273 				
274 				MathTransform transform = CRS.findMathTransform(geomCRS, kmlCRS);
275 				Geometry kmlGeometry = JTS.transform( geomGeometry, transform);
276 					
277 				Coordinate coord[] = kmlGeometry.getCoordinates();
278 				
279 				if (zAttribute != null) { //.length() != 0 ) {	// If the user has selected the height attribute
280 					FeatureType featureType = feature.getFeatureType();
281 					
282 					int attrPos = featureType.find(zAttribute);	
283 					if (attrPos == -1) {			// If the attribute doesn't exist
284 						zCoord = 0;
285 					} else {						// If the attribute exist
286 						if (!(feature.getAttribute(attrPos) instanceof Number)) {
287 							System.out.print("Height attribute isn't a number");
288 							zCoord = 0;
289 						}					
290 						if ((feature.getAttribute(attrPos) instanceof Float) || (feature.getAttribute(attrPos) instanceof Double)) {
291 							Double zCoordDouble = (Double)feature.getAttribute(attrPos); 
292 							zCoord = zCoordDouble.doubleValue();
293 						} else {
294 							Long zCoordLong = (Long)feature.getAttribute(attrPos); 
295 							zCoord = zCoordLong.longValue();
296 						}
297 						
298 					}		
299 		        	for(int j=0;j<coord.length;j++){
300 		        		kmlout.write( coord[j].x +"," + coord[j].y + "," + zCoord/scale + " "); 
301 		        	}
302 		        } else {	// If the user hasn't selected the height attribute (Geometries with 3 coordinates)
303 	        		for(int j=0;j<coord.length;j++){
304 		        		kmlout.write( coord[j].x +"," + coord[j].y + "," + coord[j].z/scale + " "); 
305 		        	} // TODO sustituir NaN en coord z por 0
306 	        	}
307 		        
308 				kmlout.write("\n</coordinates>\n");
309 				kmlout.write("</LinearRing>\n");
310 				kmlout.write("</outerBoundaryIs>\n");
311 				kmlout.write("</Polygon>\n");
312 			}	
313 			if (multiGeometry) kmlout.write("</MultiGeometry>\n");
314 			
315 			} catch (Exception e) {
316 				e.printStackTrace();
317 			}
318 		}
