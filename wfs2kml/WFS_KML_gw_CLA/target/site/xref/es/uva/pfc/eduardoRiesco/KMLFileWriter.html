23   * Translate the geographic features information into a file in KML format.
24   * 
25   * @author Eduardo Riesco

28  public class KMLFileWriter {

30  	private PrintWriter kmlout;
31  	private String kmlFileName;
32  	private String zAttribute;  // z coordinate
33  	private double scale;		// heightParameter = zAttribute/scale
34  	private String typeName;	// Feature Type Name
35  	
36  	/**
37  	 * @param kmlFileName	Name of the output KML file
38  	 * @param zAttribute	Name of the feature attribute used as height 
39  	 * @param scale			HeightParameter = zAttribute/scale
40  	 * @param typeName		Feature Type Name. 
41  	 */
42  	public KMLFileWriter(String kmlFileName, String zAttribute, double scale, String typeName) {
43  		try{
44  			this.kmlFileName = kmlFileName;
45  			this.zAttribute = zAttribute;
46  			this.scale = scale;
47  			this.typeName = typeName;
48  			FileWriter fw = new FileWriter(kmlFileName);
49  			BufferedWriter bw = new BufferedWriter(fw);
50  			kmlout = new PrintWriter (bw);
51  		}
52  		catch(IOException e)
53  		{
54  			e.printStackTrace();
55  		}
56  	}

58  	/**
59  	 * Output KML file
60  	 * 
61  	 * @param featureCollection	Features to be translated into a KML format
62  	 * @param bbox				Space region that defines the KML viewer focus. 
63  	 * 							All the features are inside of this Bounding Box
64  	 * @param geomCRS			Primary Geometry Local Name
65  	 */
66  	public void createFile(FeatureCollection featureCollection, Envelope bbox, CoordinateReferenceSystem geomCRS){  
67  		try{				
68  			kmlout.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
69  			kmlout.write("<kml xmlns=\"http://earth.google.com/kml/2.2\">\n\n");
70  			kmlout.write("<Document>\n");
71  			kmlout.write("<name>"+kmlFileName+"</name>\n");
72  			
73  			kmlout.write("\n<Folder>\n");
74  			kmlout.write("<name>"+typeName+"</name>\n"); 
75  			kmlout.write("<open>1</open>\n");
76  			kmlout.write("<description></description>\n");
77  		
78  			kmlStyle( );
79  			kmlRegion( bbox );
80  			kmlPlacemarks( featureCollection, geomCRS );
81  			
82  			kmlout.write("</Folder>\n");
83  			kmlout.write("</Document>\n");
84  			kmlout.write("</kml>\n");
85  			
86  			kmlout.close();
87  		}
88  		catch(Exception e)
89  		{
90  			e.printStackTrace();
91  		}
92  	}

134 	 * @param featureCollection	All the features to be represented
135 	 */
136 	private void kmlPlacemarks(FeatureCollection featureCollection, CoordinateReferenceSystem geomCRS) {
137 		Iterator iterator = featureCollection.iterator();			// Feature
138         try {
139             for( int f=0; iterator.hasNext(); f++) {
140                 Feature feature = (Feature) iterator.next();
141 				kmlout.write("\n<Placemark>\n");  						
142 				kmlout.write("<name>"+feature.getID() +"</name>\n");
143 				kmlout.write("<styleUrl>default</styleUrl>\n");
144 				
145 				kmlGeometries( feature, geomCRS );
146 				
147 				kmlout.write("</Placemark>\n");
148             }
149         }
150         finally {
151             	featureCollection.close( iterator );
152         }
153 	}

155 	/**
156 	 * @param feature	Feature to extract their geometries
157 	 */
158 	private void kmlGeometries(Feature feature, CoordinateReferenceSystem geomCRS) {	
159 		boolean multiGeometry = false;
160 		double zCoord;
161 		CoordinateReferenceSystem kmlCRS;
162 		
163 		try {
164 			
165 		kmlCRS = CRS.decode("EPSG:4326");
166 		if (feature.getPrimaryGeometry().getNumGeometries()>1){	
167 			kmlout.write("<MultiGeometry>\n");
168 			multiGeometry = true;
169 		}
170 		for(int g=0; g<feature.getPrimaryGeometry().getNumGeometries(); g++){  // Geometry
171 			kmlout.write("<Polygon>\n");					// 	TODO geometrias diferentes de "poligon" 
172 			kmlout.write("<extrude>1</extrude>\n");
173 			//kmlout.write("<tessellate>1</tessellate>\n");  	TODO Activar para poligonos grandes.
174 			kmlout.write("<altitudeMode>relativeToGround</altitudeMode>\n");	// TODO Otros altitudeMode
175 			
176 			kmlout.write("<outerBoundaryIs>\n");  // TODO prever InnerBoundaryIS
177 			
178 			kmlout.write("<LinearRing>\n");
179 			kmlout.write("<coordinates>\n");
180 			// JTSUtilities
181 			
182 			// zMinMax(Coordinate[] cs)
183 	        //  Determine the min and max "z" values in an array of Coordinates.
184 			
185 			// guessCoorinateDims(Coordinate[] cs)
186 //	          Returns:
187 //	        	  2 for 2d (default)
188 //	        	  4 for 3d - one of the oordinates has a non-NaN z value
189 //	        	  (3 is for x,y,m but thats not supported yet) 
190 			
191 //			findBestGeometryType(Geometry geom)
192 //	          Determine the best ShapeType for a given Geometry.
193 			
194 			Geometry geomGeometry = feature.getPrimaryGeometry().getGeometryN(g);
195 			
196 			MathTransform transform = CRS.findMathTransform(geomCRS, kmlCRS);
197 			Geometry kmlGeometry = JTS.transform( geomGeometry, transform);
198 				
199 			Coordinate coord[] = kmlGeometry.getCoordinates();
200 			
201 			if (zAttribute.length() > 0 ) {	// If the user has selected the height attribute
202 				FeatureType featureType = feature.getFeatureType();
203 				
204 				int attrPos = featureType.find(zAttribute);	
205 				if (attrPos == -1) {			// If the attribute doesn't exist
206 					zCoord = 0;
207 				}
208 				else {							// If the attribute exist
209 					if (!(feature.getAttribute(attrPos) instanceof Number)) {
210 						System.out.print("Height attribute isn't a number");
211 						zCoord = 0;
212 					}					
213 					if ((feature.getAttribute(attrPos) instanceof Float) || (feature.getAttribute(attrPos) instanceof Double)) {
214 						Double zCoordDouble = (Double)feature.getAttribute(attrPos); 
215 						zCoord = zCoordDouble.doubleValue();
216 					} else {
217 						Long zCoordLong = (Long)feature.getAttribute(attrPos); 
218 						zCoord = zCoordLong.longValue();
219 					}
220 					
221 				}		
222 	        	for(int j=0;j<coord.length;j++){
223 	        		kmlout.write( coord[j].x +"," + coord[j].y + "," + zCoord/scale + " "); 
224 	        	}
225 	        } else {	// If the user hasn't selected the height attribute (Geometries with 3 coordinates)
226         		for(int j=0;j<coord.length;j++){
227 	        		kmlout.write( coord[j].x +"," + coord[j].y + "," + coord[j].z/scale + " ");
228 	        	}
229         	}
230 	        
231 			kmlout.write("\n</coordinates>\n");
232 			kmlout.write("</LinearRing>\n");
233 			kmlout.write("</outerBoundaryIs>\n");
234 			kmlout.write("</Polygon>\n");
235 		}	
236 		if (multiGeometry) kmlout.write("</MultiGeometry>\n");
237 		
238 		} catch (Exception e) {
239 			e.printStackTrace();
240 		}
241 	}
